\begin{abstract}
We present a rigorous formulation of the Quantum Fourier Transform (QFT), including proofs of unitarity and its tensor product decomposition. These results establish the mathematical correctness of the QFT and justify its efficient circuit implementation.
\end{abstract}

\section{Preliminaries}

Let $\mathcal{H} = \mathbb{C}^{2^n}$ be a Hilbert space with computational basis
$\{\ket{0}, \dots, \ket{2^n -1}\}$.

Define:
\begin{equation}
\omega_N = e^{\frac{2\pi i}{N}}, \quad N = 2^n
\end{equation}

\section{Definition}

\textbf{Definition 1 (QFT).}
The Quantum Fourier Transform is the linear operator
$F_N : \mathcal{H} \to \mathcal{H}$ defined by:
\begin{equation}
F_N \ket{j} = \frac{1}{\sqrt{N}} \sum_{k=0}^{N-1} \omega_N^{jk} \ket{k}
\end{equation}

\section{Main Results}

\subsection{Theorem 1: Unitarity of QFT}

\textbf{Theorem.}
The operator $F_N$ is unitary, i.e.,
\begin{equation}
F_N^\dagger F_N = I
\end{equation}

\textbf{Proof.}

We compute:
\begin{align}
\langle j' | F_N^\dagger F_N | j \rangle
&= \langle F_N j' | F_N j \rangle \\
&= \frac{1}{N} \sum_{k=0}^{N-1} \omega_N^{-j'k} \omega_N^{jk} \\
&= \frac{1}{N} \sum_{k=0}^{N-1} \omega_N^{(j - j')k}
\end{align}

Let $d = j - j'$. Then:
\begin{equation}
\sum_{k=0}^{N-1} \omega_N^{dk}
=
\begin{cases}
N, & d = 0 \\
0, & d \neq 0
\end{cases}
\end{equation}

Therefore:
\begin{equation}
\langle j' | F_N^\dagger F_N | j \rangle = \delta_{j,j'}
\end{equation}

Hence:
\begin{equation}
F_N^\dagger F_N = I
\end{equation}

\hfill $\blacksquare$

\subsection{Theorem 2: Tensor Product Decomposition}

\textbf{Theorem.}
Let $j = j_1 j_2 \dots j_n$ be the binary expansion of $j$. Then:
\begin{equation}
F_N \ket{j}
=
\bigotimes_{x=1}^{n}
\frac{1}{\sqrt{2}}
\left(
\ket{0} + e^{2\pi i \, 0.j_x j_{x+1}\dots j_n} \ket{1}
\right)
\end{equation}

\textbf{Proof.}

We rewrite:
\begin{equation}
k = \sum_{x=1}^{n} k_x 2^{n-x}, \quad k_x \in \{0,1\}
\end{equation}

Then:
\begin{align}
\omega_N^{jk}
&= e^{\frac{2\pi i j}{2^n} \sum_{x=1}^{n} k_x 2^{n-x}} \\
&= \prod_{x=1}^{n} e^{2\pi i j k_x / 2^x}
\end{align}

Thus:
\begin{align}
F_N \ket{j}
&= \frac{1}{\sqrt{2^n}} \sum_{k_1,\dots,k_n}
\prod_{x=1}^{n} e^{2\pi i j k_x / 2^x}
\ket{k_1 \dots k_n} \\
&= \bigotimes_{x=1}^{n}
\frac{1}{\sqrt{2}}
\left(
\ket{0} + e^{2\pi i j / 2^x} \ket{1}
\right)
\end{align}

Finally, express:
\begin{equation}
\frac{j}{2^x} = 0.j_x j_{x+1} \dots j_n
\end{equation}

which completes the proof.

\hfill $\blacksquare$

\subsection{Theorem 3: Correctness of QFT Circuit}

\textbf{Theorem.}
The standard QFT circuit composed of Hadamard and controlled rotation gates implements $F_N$ up to qubit reversal.

\textbf{Proof (Sketch).}

From Theorem 2, each qubit output depends only on:
\begin{itemize}
    \item Its own Hadamard transformation
    \item Phase contributions from higher-order bits
\end{itemize}

Each phase term:
\begin{equation}
e^{2\pi i / 2^k}
\end{equation}
is implemented via a controlled-$R_k$ gate.

Thus, the circuit reproduces exactly the tensor decomposition of $F_N$. The reversal arises from ordering of tensor factors.

\hfill $\blacksquare$

\section{Discussion}\label{sec:discussion}

These results establish:
\begin{itemize}
    \item QFT is a unitary transformation
    \item It admits an efficient circuit decomposition
    \item The circuit structure directly follows from its algebraic form
\end{itemize}

\section{Conclusion}\label{sec:conclusion}

The Quantum Fourier Transform is rigorously characterized as a unitary operator with a structured tensor decomposition.
This decomposition directly leads to an efficient quantum circuit implementation.
